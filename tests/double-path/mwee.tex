\documentclass[tikz,border=5pt]{standalone}
\begin{document}
\begin{tikzpicture}[line join=round,line cap=round]
    \coordinate (start) at (0,1.5); 
    \coordinate (end) at (0,0);
    \path % define the points at first
        (start) -- ++(3,0) coordinate (doublestart) 
        ++(-.5,-1.5) coordinate (doubleend);
    \begin{scope}
        \clip ([xshift=5pt]doublestart) rectangle ([xshift=-5pt]doubleend);% clip the double path
        \draw[
            cyan,double,
            double distance=2.5pt,
            shorten >=-2pt,
            shorten <=-2pt
            ] 
        (doublestart) .. controls ++(.25,-.75) and ++(-.3,1) .. (doubleend)
        ;
    \end{scope}
    \draw[magenta] 
        (start) -- ([xshift=1.35pt]doublestart)
        ([xshift=1.55pt]doubleend) -- (end); 
    \end{tikzpicture}

\end{document}
